Linux 命令

栏目：Linux相关        日期：2020-07-20    阅读：

权限

用户权限

# 创建群组
groupadd {groupname}
usermod -a -G {groupname} {username}

# 授权
chmod -R 2775 {dir}
chown -R {user} {dir}
chgrp -R {groupname} {dir}

网络

端口占用

lsof -i tcp:80
lsof -nP -iTCP:6379

网络连接

netstat -tulpen

tcpdump抓包

tcpdump -i eth1 -vvv dst host www.baidu.com
tcpdump -i eth1 -w file.pcap dst host www.baidu.com
tcpdump -i eth0 -s0 -G 1800 -w %Y-%m-%d_%H%M%S.pcap host www.baidu.com #分段保存

基础语法

for循环

for i in `ls /`;do echo $i;done
for i in `ls | grep 2020-05-09`;do echo $i;done;

端口开放

## 查看已开放端口
firewall-cmd --zone=public --list-ports

## 添加开放端口
firewall-cmd --zone=public --add-port=80/tcp --permanent

## 生效配置
firewall-cmd --reload

匹配

grep -Ei  "2020-09-17 17:3[0-9]{1,}"
#匹配内容
grep -oEi "2020-09-17 17:3[0-9]{1,}"
#提取
awk -F '[_@]' '{print $2}

#json提取
jq 'select(.test=="1234")' | jq '(.test)' | sed 's/\"//g' | awk -F '@' '{print $1}'

循环

for i in `ls /`;do echo $i;done
for i in `ls | grep 2020-05-09`;do echo $i;done;

软连接

ln -s old.file soft.link

日期

date -d "${D} +${DAYS} day" +"%Y-%m-%d"

查找

find ./ -name "cx*" -atime -1 -ctime -1 -mtime -1 -exec rm {} \

高级使用

tmux

#创建
tmux new -s <session-name>
#离开
ctrl+b d

#列表
tmux ls

#接入会话
tmux attach -t <session-id>
tmux a -t <session-name>

#结束会话
tmux kill-session -t <session-id>
tmux kill-session -t <session-name>

#重命名会话
tmux rename-session -t 0 <new-name>

#多窗口
ctrl+b c
ctrl+b x
ctrl+b p
ctrl+b n

去重统计

sort test.txt | uniq -c | sort -rn
#交集
sort | uniq -d
#差集
sort | uniq -u
